1   package sokoban.kayttoliittyma;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   import sokoban.logiikka.Kartta;
6   import sokoban.logiikka.MaaObjektit;
7   
8   /**
9    * Luokka on vastuussa MaaObjektit-oliossa olevan tiedon piirtämisestä.
10   * 
11   */
12  public class MaaObjektitPiirto {
13      private MaaObjektit objektit;
14  
15      /**
16       * Luo karttaa vastaavan piirtäjän kartan MaaObjektit-oliolle.
17       * @param kartta piirrettävä kartta
18       */
19      public MaaObjektitPiirto(Kartta kartta) {
20          this.objektit = kartta.getMaaObjektit();
21      }
22      
23      /**
24       * Piirtää kartan uloskäynnin ja kytkimet. Kytkinruudut ovat vihreitä,
25       * ja uloskäynti on keltainen.
26       * 
27       * Tätä tulee kutsua vasta sen jälkeen, kun Liikkuvuus on piirretty.
28       * @param g Piirtävä grafiikkaolio
29       */
30      public void piirra(Graphics g) {
31          piirraUloskaynti(g);
32          piirraKytkimet(g);
33      }
34      
35      private void piirraUloskaynti(Graphics g) {
36          int y = objektit.getUloskayntiY();
37          int x = objektit.getUloskayntiX();
38          
39          g.setColor(Color.YELLOW);
40          g.fillRect(20 * x, 20 * y, 20, 20);
41      }
42      
43      private void piirraKytkimet(Graphics g) {
44          for (int i = 0; i < objektit.getKytkimienMaara(); i++) {
45              int y = objektit.getKytkinY(i);
46              int x = objektit.getKytkinX(i);
47              
48              g.setColor(Color.GREEN);
49              g.fillRect(20 * x, 20 * y, 20, 20);
50          }
51      }
52  }
