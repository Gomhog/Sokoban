3   import java.io.File;
4   import java.util.*;
5   
6   /**
7    * Tämän luokan tehtävänä on tuottaa käyttöliittymälle tieto valittavissa
8    * olevista kartoista.
9    * 
10   */
11  public class KarttaListaaja {
12      /**
13       * Metodi tuottaa listan merkkijonoista, jotka vastaavat level-polusta
14       * löytyviä LVL-tiedostoja.
15       * 
16       * @return Lista kaikista level-polun LVL-tiedostoista.
17       */
18      public static List<String> listaaKartat() {
19          List<String> karttaNimet = new ArrayList<>();
20          File hakemisto = new File("level");
21          if (hakemisto.list() == null) {
22              return karttaNimet;
23          }
24          for (String tiedostonimi : hakemisto.list()) {
25              if (tiedostonimi.endsWith(".lvl")) {
26                  karttaNimet.add(tiedostonimi);
27              }
28          }
29          return karttaNimet;
30      }
31      
32  }
