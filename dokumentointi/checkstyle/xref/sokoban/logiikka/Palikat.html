5   /**
6    * Luokka sisältää tiedot palikoiden sijainneista sekä niiden työntämiseen
7    * liittyvän logiikan.
8    * 
9    */
10  public class Palikat {
11      private boolean[][] sijainnit;
12  
13      /**
14       * Luo uuden palikkatiedon. Aluksi palikoita ei ole.
15       * 
16       * @param y Kentän korkeus.
17       * @param x Kentän leveys.
18       */
19      public Palikat(int y, int x) {
20          sijainnit = new boolean[y][x];
21      }
22      
23      /**
24       * Lisää palikan ruutuun.
25       * 
26       * @param y Palikan y-koordinaatti.
27       * @param x Palikan x-koordinaatti.
28       */
29      public void lisaaPalikka(int y, int x) {
30          sijainnit[y][x] = true;
31      }
32      
33      /**
34       * Kertoo, onko ruudussa palikkaa. Kentän ulkopuolella ei ole palikoita.
35       * 
36       * @param y Ruudun y-koordinaatti.
37       * @param x Ruudun x-koordinaatti.
38       * @return true, jos ruudussa on palikka, false, jos ei.
39       */
40      public boolean onkoPalikkaa(int y, int x) {
41          try {
42              return sijainnit[y][x];
43          } catch (Exception e) {
44              return false;
45          }
46      }
47      
48      /**
49       * Yrittää siirtää palikkaa annettuun suuntaan. Parametrinä annetut
50       * liikkuvuustiedot tulee antaa samasta kartasta.
51       * 
52       * @param y Halutun palikan y-koordinaatti.
53       * @param x Halutun palikan x-koordinaatti.
54       * @param suunta Haluttu siirtosuunta.
55       * @param liikkuvuus Tieto liikkumisrajoituksista
56       * @return true, jos siirto onnistuu, false, jos siirto ei onnistu.
57       */
58      public boolean siirraPalikkaa(int y, int x, Suunta suunta, 
59              Liikkuvuus liikkuvuus) {
60          if (!sijainnit[y][x]) {
61              return false;
62          }
63          int kohdeY = y + suunta.getSuuntaY();
64          int kohdeX = x + suunta.getSuuntaX();
65          if (onkoPalikkaa(kohdeY, kohdeX) || 
66                  !liikkuvuus.getPaaseekoPalikka(kohdeY, kohdeX)) {
67              return false;
68          }
69          sijainnit[y][x] = false;
70          sijainnit[kohdeY][kohdeX] = true;
71          return true;
72      }
73  }
